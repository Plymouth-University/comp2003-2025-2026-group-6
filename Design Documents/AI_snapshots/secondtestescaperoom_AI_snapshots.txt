Of course! Here is a simpler, IT-focused "Digital Escape Room" concept perfect for GCSE level.

Game Concept: "The Firewall Breach"
Brief Description: Your team are IT interns locked in the server room of "CyberCorp." A malicious hacker has activated a digital lock-down. You have 30 minutes to solve a series of IT and cyber-security puzzles to restore system control and seal the breach before all company data is deleted.

Player Roles & Simple Interdependence
The Network Tracker:

View: A network map showing connected devices (computers, servers, routers).

Puzzle: Tracks the hacker's location by following IP addresses and identifying the rogue device on the map. Uses simple logic and understanding of network layouts.

The Code Breaker:

View: A screen with lines of simple, buggy code (e.g., Python or HTML).

Puzzle: Debugs the code or identifies the correct line to unlock system functions. (e.g., Find the syntax error in a login script).

The Security Analyst:

View: A log of user activity and security alerts.

Puzzle: Spots the phishing email or the weak password that allowed the hacker in. Identifies suspicious behavior based on simple rules.

The Systems Operator:

View: The main control panel with system status (e.g., "Firewall: OFFLINE", "Backups: CORRUPTED").

Puzzle: Follows instructions from others to input codes, reboot systems in the correct sequence, and drag-and-drop security patches onto infected servers.

Interdependence Example:

The Security Analyst finds a suspicious IP address in a fake email.

They give this IP to the Network Tracker, who finds the infected computer on the map.

The Code Breaker then receives a code from the infected computer's files, but it's buggy and needs to be corrected.

Once fixed, the Systems Operator uses the corrected code to run a "virus scan" on that computer, reclaiming control.

    <script>
        // Timer functionality
        let timeLeft = 30 * 60; // 30 minutes in seconds
        const timerElement = document.getElementById('timer');
        
        function updateTimer() {
            const minutes = Math.floor(timeLeft / 60);
            const seconds = timeLeft % 60;
            timerElement.textContent = `${minutes.toString().padStart(2, '0')}:${seconds.toString().padStart(2, '0')}`;
            
            if (timeLeft <= 0) {
                clearInterval(timerInterval);
                alert("‚ùå TIME'S UP! The system has been compromised.");
            } else {
                timeLeft--;
            }
        }
        
        const timerInterval = setInterval(updateTimer, 1000);
        
        // Node selection
        let selectedNode = null;
        
        function selectNode(node) {
            // Remove selection from all nodes
            document.querySelectorAll('.network-node').forEach(n => {
                n.style.border = '1px solid #4444ff';
                n.style.boxShadow = 'none';
            });
            
            // Add selection to clicked node
            node.style.border = '2px solid #ffff00';
            node.style.boxShadow = '0 0 10px rgba(255, 255, 0, 0.7)';
            selectedNode = node;
            
            addMessage(`Selected: ${node.textContent}`, 'system');
        }
        
        // Button actions
        function quarantineNode() {
            if (selectedNode) {
                selectedNode.style.background = '#663300';
                selectedNode.style.borderColor = '#ff9900';
                selectedNode.innerHTML += ' [QUARANTINED]';
                addMessage(`Quarantined: ${selectedNode.textContent}`, 'system');
            } else {
                alert("Please select a node first!");
            }
        }
        
        function scanNode() {
            if (selectedNode) {
                addMessage(`Scanning: ${selectedNode.textContent}...`, 'system');
                // Simulate scanning delay
                setTimeout(() => {
                    if (selectedNode.classList.contains('infected')) {
                        addMessage(`Scan complete: MALWARE DETECTED on ${selectedNode.textContent}`, 'system');
                    } else {
                        addMessage(`Scan complete: No threats detected on ${selectedNode.textContent}`, 'system');
                    }
                }, 1500);
            } else {
                alert("Please select a node first!");
            }
        }
        
        function deployFirewall() {
            document.querySelector('.status-item.offline').textContent = 'üü¢ Firewall: ACTIVE';
            document.querySelector('.status-item.offline').classList.remove('offline');
            document.querySelector('.status-item.offline').style.background = '#003300';
            document.querySelector('.status-item.offline').style.color = '#66ff66';
            addMessage('Firewall deployed successfully!', 'system');
        }
        
        function restoreBackups() {
            document.querySelector('.status-item.corrupted').textContent = 'üü¢ Backups: RESTORED';
            document.querySelector('.status-item.corrupted').classList.remove('corrupted');
            document.querySelector('.status-item.corrupted').style.background = '#003300';
            document.querySelector('.status-item.corrupted').style.color = '#66ff66';
            addMessage('Backups restored successfully!', 'system');
        }
        
        function unlockRoom() {
            const firewallActive = document.querySelector('.status-item.offline') === null;
            const backupsRestored = document.querySelector('.status-item.corrupted') === null;
            
            if (firewallActive && backupsRestored) {
                clearInterval(timerInterval);
                alert('üéâ CONGRATULATIONS! You have secured the system and unlocked the server room!');
            } else {
                alert('You must deploy the firewall and restore backups before unlocking the server room!');
            }
        }
        
        // Chat functionality
        function addMessage(text, type) {
            const chatMessages = document.getElementById('chatMessages');
            const message = document.createElement('div');
            message.className = `message ${type}`;
            message.textContent = text;
            chatMessages.appendChild(message);
            chatMessages.scrollTop = chatMessages.scrollHeight;
        }
        
        // Initialize by selecting the infected server
        window.onload = function() {
            const infectedNode = document.querySelector('.network-node.infected');
            selectNode(infectedNode);
        };
